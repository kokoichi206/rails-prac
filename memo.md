```sh
# 4.0.0 以上で 4.1.0 より下のもの: マイナーアップグレード以上を許さない
gem 'coffee-rails', '~> 4.0.0'
# 4.0.0 以上で 5.0.0 より下のもの: メジャーアップデートを許さない
gem 'coffee-rails', '~> 4.0'
```
